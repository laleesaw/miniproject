Expected Task

  1. Take the fingerprint data obtained from the scan and verify it against the database.
  2. Confirm the user's fingerprint stored in the database.
    • Display the date and time when the user successfully authenticated.
    • Display the user's unique ID number.
    • Display the username of the user after successful authentication.
    • When fingerprint authentication is successful, trigger an OTP system.
  3. Display the user's information on Home Assistant.
  4. Notify the administrator when there is a failed authentication attempt.

Report on 26 September 2024

After we found troubleshooting on 26 September, We decided to change the board from Wemos lolin32 lite to Arduino Uno R3 to fix the troubleshooting. Then when we uploaded it on the Arduino Uno R3 board, a board could detect a finger sensor and when we searched for a solution, we found that the voltage supply of Wemos lolin32 lite wasn't enough, the fingerprint sensor wanted 5 Voltage but the Wemos lolin32 lite could give 3.3 Voltage. Actually, from the datasheet of AS608, we can normally use 3.3 Voltage but in practical Wemos lolin32 lite supply voltage is insufficient.

The Arduino Uno R3 is unsuitable due to its lack of built-in Wi-Fi capabilities, which are essential for projects requiring Wi-Fi connectivity, such as integrating a fingerprint sensor with Home Assistant. While it is possible to add Wi-Fi functionality to the Arduino Uno using a Wi-Fi shield, a more optimal solution is to utilize a board like the ESP32 or the Wemos Lolin32, which comes with built-in Wi-Fi. Consequently, we have opted to employ the Wemos Lolin32 for its integrated Wi-Fi and ability to provide the necessary 5-volt supply.

Here is the Example enroll code from Adafruit Fingerprint Sensor Library:

#include <Adafruit_Fingerprint.h>

// Use hardware serial on the Wemos Lolin32 (ESP32)
#define RX_PIN 16  // GPIO16 to TX (white wire)
#define TX_PIN 17  // GPIO17 to RX (yellow wire)

HardwareSerial mySerial(1);  // Use Serial1 for communication with the fingerprint sensor

Adafruit_Fingerprint finger = Adafruit_Fingerprint(&mySerial);   // Initialize fingerprint sensor

uint8_t id; // Variable to store fingerprint ID

void setup()
{
  Serial.begin(115200);  // For debugging
  mySerial.begin(57600, SERIAL_8N1, RX_PIN, TX_PIN);  // Start Serial1 for fingerprint communication

  delay(100);  // Wait for fingerprint sensor to initialize
  Serial.println("\n\nAdafruit Fingerprint sensor enrollment");

  // Verify if the fingerprint sensor is connected
  if (finger.verifyPassword()) {
    Serial.println("Found fingerprint sensor!");
  } else {
    Serial.println("Did not find fingerprint sensor :(");
    while (1) {
      delay(1);
    }  // Stay in loop if sensor is not found
  }

 // Retrieve and display fingerprint sensor parameters
  Serial.println(F("Reading sensor parameters"));
  finger.getParameters();
  Serial.print(F("Status: 0x")); Serial.println(finger.status_reg, HEX);
  Serial.print(F("Sys ID: 0x")); Serial.println(finger.system_id, HEX);
  Serial.print(F("Capacity: ")); Serial.println(finger.capacity);
  Serial.print(F("Security level: ")); Serial.println(finger.security_level);
  Serial.print(F("Device address: ")); Serial.println(finger.device_addr, HEX);
  Serial.print(F("Packet len: ")); Serial.println(finger.packet_len);
  Serial.print(F("Baud rate: ")); Serial.println(finger.baud_rate);
}

uint8_t readnumber(void) {
  uint8_t num = 0;
  // Wait for user to input a valid number through the serial monitor
  while (num == 0) {
    while (!Serial.available());
    num = Serial.parseInt();
  }
  return num;
}

void loop() {
  Serial.println("Ready to enroll a fingerprint!");
  Serial.println("Please type in the ID # (from 1 to 127) you want to save this finger as...");
  id = readnumber();  // Get the ID number from the user input
  if (id == 0) { // ID #0 not allowed, try again!
    return;
  }
  Serial.print("Enrolling ID #");
  Serial.println(id);

  // Start the fingerprint enrollment process
  while (!getFingerprintEnroll());
}

uint8_t getFingerprintEnroll() {
  int p = -1;
    
  // Wait for the user to place a valid finger on the sensor
  Serial.print("Waiting for valid finger to enroll as #"); Serial.println(id);
  while (p != FINGERPRINT_OK) {
    p = finger.getImage();  // Capture fingerprint image
    switch (p) {
    case FINGERPRINT_OK:
      Serial.println("Image taken");
      break;
    case FINGERPRINT_NOFINGER:
      Serial.print(".");  // Indicate no finger is detected
      break;
    case FINGERPRINT_PACKETRECIEVEERR:
      Serial.println("Communication error");
      break;
    case FINGERPRINT_IMAGEFAIL:
      Serial.println("Imaging error");
      break;
    default:
      Serial.println("Unknown error");
      break;
    }
  }

  // Convert image to fingerprint template (slot 1)
  p = finger.image2Tz(1);
  switch (p) {
    case FINGERPRINT_OK:
      Serial.println("Image converted");
      break;
    case FINGERPRINT_IMAGEMESS:
      Serial.println("Image too messy");
      return p;
    case FINGERPRINT_PACKETRECIEVEERR:
      Serial.println("Communication error");
      return p;
    case FINGERPRINT_FEATUREFAIL:
      Serial.println("Could not find fingerprint features");
      return p;
    case FINGERPRINT_INVALIDIMAGE:
      Serial.println("Could not find fingerprint features");
      return p;
    default:
      Serial.println("Unknown error");
      return p;
  }

  Serial.println("Remove finger");
  delay(2000);  // Wait for the user to remove the finger
  
  // Wait until the finger is removed
  p = 0;
  while (p != FINGERPRINT_NOFINGER) {
    p = finger.getImage();
  }
  Serial.print("ID "); Serial.println(id);
  // Wait for the user to place the same finger again
  p = -1;
  Serial.println("Place same finger again");
  while (p != FINGERPRINT_OK) {
    p = finger.getImage();
    switch (p) {
    case FINGERPRINT_OK:
      Serial.println("Image taken");
      break;
    case FINGERPRINT_NOFINGER:
      Serial.print(".");
      break;
    case FINGERPRINT_PACKETRECIEVEERR:
      Serial.println("Communication error");
      break;
    case FINGERPRINT_IMAGEFAIL:
      Serial.println("Imaging error");
      break;
    default:
      Serial.println("Unknown error");
      break;
    }
  }


  // Convert the second image to template (slot 2)
  p = finger.image2Tz(2);
  switch (p) {
    case FINGERPRINT_OK:
      Serial.println("Image converted");
      break;
    case FINGERPRINT_IMAGEMESS:
      Serial.println("Image too messy");
      return p;
    case FINGERPRINT_PACKETRECIEVEERR:
      Serial.println("Communication error");
      return p;
    case FINGERPRINT_FEATUREFAIL:
      Serial.println("Could not find fingerprint features");
      return p;
    case FINGERPRINT_INVALIDIMAGE:
      Serial.println("Could not find fingerprint features");
      return p;
    default:
      Serial.println("Unknown error");
      return p;
  }


  // Create fingerprint model from the two captured images
  Serial.print("Creating model for #");  Serial.println(id);

  p = finger.createModel();
  if (p == FINGERPRINT_OK) {
    Serial.println("Prints matched!");
  } else if (p == FINGERPRINT_PACKETRECIEVEERR) {
    Serial.println("Communication error");
    return p;
  } else if (p == FINGERPRINT_ENROLLMISMATCH) {
    Serial.println("Fingerprints did not match");
    return p;
  } else {
    Serial.println("Unknown error");
    return p;
  }

  // Store the fingerprint model in the specified ID
  Serial.print("ID "); Serial.println(id);
  p = finger.storeModel(id);
  if (p == FINGERPRINT_OK) {
    Serial.println("Stored!");
  } else if (p == FINGERPRINT_PACKETRECIEVEERR) {
    Serial.println("Communication error");
    return p;
  } else if (p == FINGERPRINT_BADLOCATION) {
    Serial.println("Could not store in that location");
    return p;
  } else if (p == FINGERPRINT_FLASHERR) {
    Serial.println("Error writing to flash");
    return p;
  } else {
    Serial.println("Unknown error");
    return p;
  }

  return true;
}

Build a WiFi connected Fingerprint Sensor with Home Assistant

Hardware

Hardware list

  1. Bread Board
  2. Jump Wire
  3. AS608 fingerprint sensor
  4. ESP32 Wemos lolin32
    • You should be able to use any board that provides the 5 Voltage supply along with built-in Wi-Fi
  5. Raspberry pi 4
    • We use a Raspberry Pi 4 to run Home Assistant. Because the Raspberry Pi 4 has enough power to handle Home Assistant and other services like the Mosquitto MQTT broker smoothly.

We have integrated the AS608 optical fingerprint sensor, which is designed to scan fingerprints and transmit the processed data to a microcontroller via serial communication. This module securely stores all registered fingerprints and has the capacity to accommodate up to 120 individual fingerprint records. The all-in-one optical fingerprint sensor facilitates seamless integration for fingerprint detection and verification.

And here is some of the specifications of this sensor from the datasheet:

  • Name: AS608 Optical Fingerprint Reader Module.
  • Operation Voltage: (3.3~5)Vdc.
  • Interface: TTL Serial.
  • Baud rate: (9600~57600) (default 57600).
  • Rated Current: ~120mA.
  • Fingerprint imaging time: <1.0 seconds.
  • Storage capacity: 162 templates.
  • Template file: 512 bytes.
  • False Acceptance Rate: <0.001% (Security level 3).
  • False Reject Rate: <1.0% (Security level 3).
  • Safety Level: 1~5 low to high safety.
  • Temperature: -20 to +50 degrees.

Node-RED and Automation

There are 2 flows that we are using

The first flow act as an event trigger that will trigger everytime a fingerprint is "matched" or "unmatched".
The second flow plays the role of generating OTP and receiving OTP to trigger light automation.

Input valid key is from a helper of HA(Home Assistant) that will generate a random number every 30 second called random sensor. And the valid key is a number selected when there are a "finger matched" event. We set the current number as a valid key and send it as the correct number for the OTP interactive message notification. Then after 4 cycle of generating number(4 time 30 second) the valid key becomes invalid.

This is the code of notif create block as you can see, it recieves the valid key as msg.payload and create 2 random numbers then put the right and wrong nmuber in random position. If user press the correct button the notification will send "CORRECT_NUMBER" action and if user press wrong number it will send "WRONG_NUMBER" action. These two action will be listened by a listener in the Home Assistant and call it an event.

code:

// Get the correct number from the incoming msg.payload
let X = msg.payload;

// Prepare the JSON data structure for the notification
let jsonData = {
    "title": "Action Notification Test",
    "message": "Press the correct number.",
    "data": {
        "actions": [
            {
                "action": "WRONG_NUMBER", // Fixed action identifier
                "title": "Wrong Number 1"    // Placeholder for random number
            },
            {
                "action": "WRONG_NUMBER", // Fixed action identifier
                "title": "Wrong Number 2"    // Placeholder for random number
            },
            {
                "action": "CORRECT_NUMBER",  // This will be the correct number
                "title": ${Math.floor(X)}             // Dynamic correct number from msg.payload
            }
        ]
    }
};

// Function to generate random two-digit numbers between 10 and 99
function generateRandomTwoDigit() {
    return Math.floor(Math.random() * 90) + 10;
}

// Generate two random numbers for the wrong options
let randomNumbers = [generateRandomTwoDigit(), generateRandomTwoDigit()];

// Set the random numbers in the title for wrong actions
jsonData.data.actions[0].title = ${randomNumbers[0]};  // First wrong number
jsonData.data.actions[1].title = ${randomNumbers[1]};  // Second wrong number

// Shuffle function to randomize the order of actions
function shuffle(array) {
    for (let i = array.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [array[i], array[j]] = [array[j], array[i]];
    }
}

// Shuffle the actions list
shuffle(jsonData.data.actions);
let jsonResult = JSON.stringify(jsonData, null, 4);

// Assign the JSON object to msg.payload (not as a string)
msg.notification = jsonData;
msg.payload = jsonResult;

// Return the msg object to pass it along the flow
return msg;

As the even occur the switch will catch the event.

The switch will works as the HA event listener can listen whether user press the wrong or correct button.

The light 50 and light 25 will be the trigger of these automation.
The light 50 is green light indicate that correct number is pressed and red(light 25) is for wrong number.

Troubleshooting

  1. As always, be sure to check wiring, in particular the TX and RX pins are the correct way round.
  2. Ensure that you are getting power to the sensor, there are testing pads on the back that you can check the voltage, you should be receiving +3.3~5V at the pads.
